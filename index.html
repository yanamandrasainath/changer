from selenium import webdriver from selenium.webdriver.common.by import By from selenium.webdriver.support.ui import WebDriverWait, Select from selenium.webdriver.support import expected_conditions as EC import streamlit as st st.set_page_config( page_title="Unified E-Government Portal", page_icon="🏛️", layout="wide" ) st.title("🏛️ Unified E-Government Portal") st.markdown("---") # Layout left, right = st.columns([1, 3]) # LEFT SIDE MENU with left: st.subheader("Services") service = st.radio( "", [ "Scholarship", "MeeSeva Status" ] ) # RIGHT SIDE CONTENT with right: if service == "Scholarship": st.subheader("Scholarship Status") app_number = st.text_input( "Application Number" ) year = st.selectbox( "Academic Year", [ "2024-25", "2023-24", "2022-23" ] ) if st.button("Get Status"): if app_number == "": st.error("Please Enter Application Number") else: with st.spinner("Opening Scholarship Portal..."): try: driver = webdriver.Chrome() driver.get( "https://tgepass.cgg.gov.in/HomeServicePostmatricKnowApplication" ) wait = WebDriverWait(driver, 10) inputs = wait.until( EC.presence_of_all_elements_located( (By.TAG_NAME, "input") ) ) inputs[0].send_keys(app_number) try: select = Select( driver.find_element( By.TAG_NAME, "select" ) ) select.select_by_visible_text(year) except: st.error("Dropdown Not Found") try: get_status_btn = driver.find_element( By.XPATH, "//*[contains(text(),'Get Status')]" ) get_status_btn.click() except: try: get_status_btn = driver.find_element( By.XPATH, "//input[contains(@value,'Get Status')]" ) get_status_btn.click() except: st.warning( "Get Status Button Not Found" ) st.success( "Request Submitted Successfully" ) except Exception as e: st.error(str(e)) elif service == "MeeSeva Status": st.subheader("MeeSeva Application Status") app_no = st.text_input( "Application Number" ) if st.button("Get Data"): if app_no == "": st.error( "Please Enter Application Number" ) else: try: st.info( "Opening MeeSeva Portal..." ) driver = webdriver.Chrome() driver.maximize_window() driver.get( "https://ts.meeseva.telangana.gov.in/meeseva/home.htm" ) wait = WebDriverWait( driver, 20 ) # Find Application Status Field status_box = wait.until( EC.element_to_be_clickable( ( By.ID, "searchAppl" ) ) ) status_box.clear() status_box.send_keys( app_no ) st.success( "Application Number Filled" ) # Click Search Icon driver.execute_script( "goApplicationStatus();" ) st.success( "Search Submitted Successfully" ) except Exception as e: st.error( str(e) )
